# `std`

`std` - *пространство имён* стандартной библиотеки
Лезть в `std` *права не имеем*

# Считывание
---
```c++
#include <iostream>
#include <print>

//using namespace std; //*так делать не будем*

int main()
{
    int x;
    std::cin >> x; //здесь операция >> *перегружена*

    using std::cin; //теперь можно не писать std у cin
    long long y;
    cin >> y;
}
```
---

## `using std::cin` вне main
---
```c++
#include <iostream>
#include <print>

using std::cin;

int main()
{
    int x;
    cin >> x;
}
```

# Си в C++
---
```c++
#include <iostream>
#include <print>
#include <cstring> //Сишная библиотека

int main()
{
    int x;
    std::cin >> x;

    using std::cin;
    long long y;
    cin >> y;

    std::memset(&x, 0, sizeof(x));
    
    //либо

    memset(&x, 1, sizeof(x)); //т.е. можно не использовать `std`
}
```

# Считывание + проверка на конец файла
---
```c++

{
    int x;
    int s{}; //инициализация 0

    while(std::cin >> x){
        s += x;
    }

    std::println("{}", s);
}
```

## 2 программа
---
```c++
#include <print>
#include <iostream>

int main()
{
    int x;
    int s{}; //инициализация 0

/*
    while(std::cin >> x){
        s += x;
    }
*/

    while(1){
        auto &t = std::cin >> x;
        std::println("{}", std::is_same_v<decltype(t), std::istream&>);
        auto b = static_cast<bool>(t);
        //bool b = bool(t);
        //bool b{t};
        if(!b) break;
        std::println("{}", x);
    }

    std::println("{}", s);
}
```

# Считывание символов с пробелами
---
```c++
#include <iostream>
#include <print>

int main()
{
    char c;
    
    std::cin >> std::noskipws; //манипулятор, который перестаёт игнорировать переменные
    while(std::cin >> c){
        std::println("{}", c);
    }
    std::cin >> std::skipws;
}
```
---

## 2 программа
---
```c++
#include <iostream>
#include <print>

int main()
{
    char c;
    
    while(std::cin.get(c)){ //также не игнорирует побелы. Так предпочтительнее делать
        std::println("{}", c);
    }
}
```
